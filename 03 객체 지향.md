### 생성자

```c#
using System;

namespace CSharp
{
    class Player
    {
        public string nickname;
        public int hp;
        public int attack;

        // 생성자
        public Player()
        {
            this.nickname = "닉네임없음";
            this.hp = 100;
            this.attack = 10;
            Console.WriteLine("기본 생성자 호출!");
        }

        // 생성자 오버로딩
        public Player(string nickname) : this()
        {
            this.nickname = nickname;
            Console.WriteLine ("nickname 생성자 호출!");
        }

        public Player(string nickname, int hp, int attack) : this(nickname)
        {
            this.hp = hp;
            this.attack = attack;
            Console.WriteLine("nickname hp attack 생성자 호출!");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            Player player1 = new Player();
            Console.WriteLine();
            // 기본 생성자 호출!

            Player player2 = new Player("cherry");
            Console.WriteLine();
            // 기본 생성자 호출!
            // nickname 생성자 호출!

            Player player3 = new Player("cherry", 200, 20);
            Console.WriteLine();
            // 기본 생성자 호출!
            // nickname 생성자 호출!
            // nicknmae hp attack 생성자 호출!
        }
    }
}
```

### static

```c#
using System;

namespace CSharp
{
    class Knight
    {
        // 정적 필드
        static int counter = 0;

        public int id;
        public int hp;
        public int attack;

        public Knight()
        {
            this.id = counter++;
            this.hp = 100;
            this.attack = 10;
            Console.WriteLine("생성자 호출!");
        }

        // 정적 메서드
        static public int GetCounter()
        {
            return counter;
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            Knight knight = new Knight();  // 생성자 호출!
            Console.WriteLine($"{Knight.GetCounter()}");  // 1
        }
    }
}
```

### base - 부모를 나타내는 키워드

```c#
using System;

namespace CSharp
{
    // 부모 클래스
    class Player
    {
        public string name;

        public Player()
        {
            this.name = "이름없음";
            Console.WriteLine("Player 생성자 호출!");
        }

        public Player(string name)
        {
            this.name = name;
            Console.WriteLine("Player name 생성자 호출!");
        }

        public void Move()
        {
            Console.WriteLine("Player 이동!");
        }
    }

    // 자식 클래스
    class Mage : Player
    {
        public int mp = 100;

        public Mage() : base()
        {
            Console.WriteLine("Mage 생성자 호출!");
        }

        public Mage(string name) : base(name)
        {
            Console.WriteLine("Mage name 생성자 호출!");
        }

        public void Teleport()
        {
            Console.Write("텔레포트 ");
            base.Move();
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            Mage mage = new Mage("cherry");
            // Player name 생성자 호출!
            // Mage name 생성자 호출!

            mage.Teleport();  // 텔레포트 Player 이동!
        }
    }
}
```

### 은닉성 - 접근 제한자

```c#
using System;

namespace CSharp
{
    class Player
    {
        public string id;
        protected int hp;
        protected int attack;
        private string pwd;

        public Player(string id, string pwd)
        {
            this.id = id;
            this.hp = 100;
            this.attack = 10;
            this.pwd = pwd;
        }
    }

    class Knight : Player
    {
        public Knight(string id, string pwd) : base(id, pwd) { }

        public void Test()
        {
            Console.WriteLine(base.id);
            Console.WriteLine(base.hp);
            Console.WriteLine(base.attack);
            // Console.WriteLine(base.pwd);  접근 불가능!
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            Knight knight = new Knight("cherry", "1234");
            // Console.WriteLine(knight.id);  접근 불가능!
            // Console.WriteLine(knight.hp);  접근 불가능!
            // Console.WriteLine(knight.attack);  접근 불가능!
            // Console.WriteLine(knight.pwd);  접근 불가능
        }
    }
}
```

### is

```c#
using System;

namespace CSharp
{
    class Player { }

    class Knight : Player { public int energy; }
    class Mage : Player { public int mp; }

    class Program
    {
        static void Main(string[] args)
        {
            Knight knight = new Knight();
            EnterGame(knight);  // 기력 회복!

            Mage mage = new Mage();
            EnterGame(mage);  // 마나 회복!
        }

        static void EnterGame(Player player)
        {
            bool isMage = player is Mage;
            if(isMage)
            {
                Mage mage = (Mage)player;
                mage.mp = 50;
                Console.WriteLine("마나 초기화");
            }

            bool isKnight = player is Knight;
            if(isKnight)
            {
                Knight knight = (Knight)player;
                knight.energy = 20;
                Console.WriteLine("기력 회복!");
            }
        }
    }
}
```

### as

```c#
using System;

namespace CSharp
{
    class Player { }

    class Knight : Player { public int energy; }
    class Mage : Player { public int mp; }

    class Program
    {
        static void Main(string[] args)
        {
            Knight knight = new Knight();
            EnterGame(knight);  // 기력 회복!

            Mage mage = new Mage();
            EnterGame(mage);  // 마나 회복!

        static void EnterGame(Player player)
        {
            Mage mage = player as Mage;
            if(mage != null)
            {
                mage.mp = 50;
                Console.WriteLine("마나 초기화!");
            }

            Knight knight = player as Knight;
            if(knight != null)
            {
                knight.energy = 20;
                Console.WriteLine("기력 회복!");
            }
        }
    }
}
```

### 다형성 - 오버라이딩(new)

```c#
using System;

namespace CSharp
{
    class Player
    {
        public void Move()
        {
            Console.WriteLine("Player 이동!");
        }
    }

    class Knight : Player
    {
        public new void Move()
        {
            Console.WriteLine("Knight 이동!");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            Knight knight = new Knight();
            EnterGame(knight);  // Knight 이동!
        }

        static void EnterGame(Player player)
        {
            Knight knight = player as Knight;
            if (knight != null)
                knight.Move();
        }
    }
}
```

### 다형성 - 오버라이딩(virtual & override)

```c#
using System;

namespace CSharp
{
    class Player
    {
        public virtual void Move()
        {
            Console.WriteLine("Player 이동!");
        }
    }

    class Knight : Player
    {
        public override void Move()
        {
            Console.WriteLine("Knight 이동!");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            Knight knight = new Knight();
            EnterGame(knight);  // Knight 이동!
        }

        static void EnterGame(Player player)
        {
            player.Move();
        }
    }
}
```
